    ```typescript
    export const adminProcedure = t.procedure.use(({ ctx, next }) => {
      if (!ctx.user) {
        throw new TRPCError({
          code: 'UNAUTHORIZED',
          message: 'ログインが必要です',
        })
      }
      if (ctx.user.role !== 'admin') {
        throw new TRPCError({
          code: 'FORBIDDEN',
          message: '管理者権限が必要です',
        })
      }
      return next({
        ctx: { ...ctx, user: ctx.user },
      })
    })
    ```
